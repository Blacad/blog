def cascade(n):
    if n<10:
        print(n)
    else:
        print(n)
        cascade(n//10)
        print(n)

def inverse_cascade(n):
    grow(n)
    print(n)
    shrink(n)

def fg(f,g,n):
    if n:
        f(n)
        g(n)


grow = lambda n: fg(grow, print, n//10)
shrink = lambda n: fg(print,shrink, n//10)

cascade(12345)
print()
inverse_cascade(12345)
