def split(n):
    return n // 10,n % 10

def sum_digits(n):
    if n < 10:
        return n
    else:
        rest , last = split(n)
        return sum_digits(rest)+last
def lunh_sum(n):
    if n < 10:
        return n
    else:
        rest , last = split(n)
        return lunh_double_sum(rest)+last
def lunh_double_sum(n):
    rest,last = split(n)
    digit = sum_digits(2*last)
    if n< 10:
        return n
    else:
        return lunh_sum(rest)+digit
